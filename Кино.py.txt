import numpy as np 
import pandas as pd
import matplotlib.pyplot as plt 

np.random.seed(42)
n_movies = 50 
movie_titles = [fФильм {i} for i in range(1, n_movies + 1)] 
genres = ['Боевик', 'Драма', 'Комедия', 'Фантастика', 'Ужасы', 'Мелодрама']
movie_genres = np.random.choice(genres, size=n_movies)
budgets = np.random.uniform(20, 200, size=n_movies).round(1) 
box_office = budgets  np.random.uniform(1.5, 5, size=n_movies).round(1)
ratings = np.random.normal(7, 1.5, size=n_movies).clip(1, 10).round(1)
reviews_count = np.random.randint(10000, 500000, size=n_movies)

df = pd.DataFrame({
    'Название' movie_titles,
    'Жанр' movie_genres,
    'Бюджет (млн $)' budgets,
    'Кассовые сборы (млн $)' box_office,
    'Рейтинг' ratings,
    'Количество отзывов' reviews_count
})
df['Коэффициент окупаемости'] = (df['Кассовые сборы (млн $)']  df['Бюджет (млн $)']).round(2)
print(Данные о фильмах 2025 года)

print(top_movies[['Название', 'Рейтинг', 'Количество отзывов', 'Коэффициент окупаемости']])
genre_stats = df.groupby('Жанр').agg({
    'Рейтинг' 'mean',
    'Коэффициент окупаемости' 'mean',

}).round(2)
