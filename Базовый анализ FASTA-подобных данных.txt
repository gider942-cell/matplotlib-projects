import pandas as pd
import numpy as np
import matplotlib.pyplot as plt

# Создаём тестовый датасет
data = {
    'sequence_id': ['seq1', 'seq2', 'seq3', 'seq4'],
    'sequence': ['ATGCGTACGTA', 'ATGCGTA', 'GCTAGCTAGCTA', 'ATGC'],
    'length': [11, 7, 12, 4],
    'gc_content': [0.45, 0.57, 0.50, 0.50]
}

df = pd.DataFrame(data)
print("Исходные данные:")
print(df)

# Анализ: фильтруем последовательности по длине и GC?содержанию
filtered_df = df[(df['length'] > 5) & (df['gc_content'] > 0.4)]
print("\nОтфильтрованные последовательности (длина > 5, GC > 40%):")
print(filtered_df)


# --- ДОПОЛНИТЕЛЬНЫЕ ФУНКЦИИ ---

# 1. Функция для расчёта GC?содержания по последовательности
def calculate_gc_content(sequence):
    """Рассчитывает GC?содержание для заданной последовательности ДНК."""
    g_count = sequence.count('G')
    c_count = sequence.count('C')
    total_count = len(sequence)
    if total_count == 0:
        return 0
    return (g_count + c_count) / total_count

# Применяем функцию к исходным данным для проверки
df['calculated_gc'] = df['sequence'].apply(calculate_gc_content)
print("\nПроверка расчёта GC?содержания:")
print(df[['sequence_id', 'sequence', 'gc_content', 'calculated_gc']])

# 2. Функция для классификации последовательности по GC?содержанию
def classify_gc_content(gc_value):
    """Классифицирует последовательность по уровню GC?содержания."""
    if gc_value < 0.4:
        return 'Низкое (<40%)'
    elif gc_value <= 0.6:
        return 'Среднее (40–60%)'
    else:
        return 'Высокое (>60%)'

df['gc_class'] = df['gc_content'].apply(classify_gc_content)
print("\nКлассификация по GC?содержанию:")
print(df[['sequence_id', 'gc_content', 'gc_class']])

# 3. Статистический анализ
print("\nСтатистическая сводка:")
print(f"Средняя длина: {df['length'].mean():.2f}")
print(f"Среднее GC?содержание: {df['gc_content'].mean():.3f}")
print(f"Максимальная длина: {df['length'].max()}")
print(f"Минимальное GC?содержание: {df['gc_content'].min():.3f}")

# --- ВИЗУАЛИЗАЦИЯ ---

# Настраиваем размер графиков
plt.figure(figsize=(15, 10))

# График 1: Длина последовательностей
plt.subplot(2, 3, 1)
plt.bar(df['sequence_id'], df['length'], color='skyblue')
plt.title('Длина последовательностей')
plt.xlabel('ID последовательности')
plt.ylabel('Длина')
plt.xticks(rotation=45)

# График 2: GC?содержание последовательностей
plt.subplot(2, 3, 2)
plt.bar(df['sequence_id'], df['gc_content'], color='lightgreen')
plt.title('GC?содержание последовательностей')
plt.xlabel('ID последовательности')
plt.ylabel('GC?содержание')
plt.xticks(rotation=45)

# График 3: Распределение GC?классов
plt.subplot(2, 3, 3)
gc_class_counts = df['gc_class'].value_counts()
plt.pie(gc_class_counts.values, labels=gc_class_counts.index, autopct='%1.1f%%', startangle=90)
plt.title('Распределение по классам GC?содержания')

# График 4: Соотношение длины и GC?содержания (точечный график)
plt.subplot(2, 3, 4)
colors = ['red' if gc > 0.5 else 'blue' for gc in df['gc_content']]
plt.scatter(df['length'], df['gc_content'], c=colors, s=100, alpha=0.7)
plt.title('Соотношение длины и GC?содержания')
plt.xlabel('Длина последовательности')
plt.ylabel('GC?содержание')
for i, seq_id in enumerate(df['sequence_id']):
    plt.annotate(seq_id, (df['length'][i], df['gc_content'][i]), xytext=(5, 5),
                textcoords='offset points', fontsize=9)

# График 5: Гистограмма распределения длин
plt.subplot(2, 3, 5)
plt.hist(df['length'], bins=5, color='orange', alpha=0.7, edgecolor='black')
plt.title('Распределение длин последовательностей')
plt.xlabel('Длина')
plt.ylabel('Количество')

# График 6: Box plot для GC?содержания
plt.subplot(2, 3, 6)
plt.boxplot(df['gc_content'], vert=True, patch_artist=True, boxprops=dict(facecolor='lightcoral'))
plt.title('Box plot: GC?содержание')
plt.ylabel('GC?содержание')

# Автоматически настраиваем расположение графиков, чтобы они не перекрывались
plt.tight_layout()

# Показываем все графики
plt.show()

# Дополнительная визуализация: сравнение исходных и отфильтрованных данных
plt.figure(figsize=(12, 5))

# Исходные данные
plt.subplot(1, 2, 1)
plt.bar(df['sequence_id'], df['length'], color='gray', alpha=0.5, label='Исходные')
plt.title('Исходные последовательности: длина')
plt.xlabel('ID последовательности')
plt.ylabel('Длина')
plt.xticks(rotation=45)
plt.legend()

# Отфильтрованные данные
plt.subplot(1, 2, 2)
if not filtered_df.empty:
    plt.bar(filtered_df['sequence_id'], filtered_df['length'], color='green', alpha=0.8, label='Отфильтрованные')
    plt.title('Отфильтрованные последовательности: длина (>5, GC>40%)')
else:
    plt.text(0.5, 0.5, 'Нет данных после фильтрации', ha='center', va='center', fontsize=12)
    plt.axis('off')
plt.xlabel('ID последовательности')
plt.ylabel('Длина')
plt.xticks(rotation=45)
plt.legend()

plt.tight_layout()
plt.show()
