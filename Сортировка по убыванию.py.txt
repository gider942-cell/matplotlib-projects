import matplotlib.pyplot as plt 
import numpy as np

def plot_sorted_bars(categories, values, colors, kwargs)
    sorted_indices = np.argsort(values)[-1]
    categories_sorted = [categories[i] for i in sorted_indices]
    values_sorted = values[sorted_indices]
    
    bars = plt.bar(categories_sorted, values_sorted, color=colors, kwargs)
    return bars, categories_sorted, values_sorted
